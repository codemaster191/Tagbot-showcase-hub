This Python code finds the largest and smallest of three numbers, and checks if any are equal:

x = int(input("Enter a number ")) 
y = int(input("Enter another number ")) 
z = int(input("Enter another number ")) 

if x < y and x < z:
    smallest = x 
elif y < z:
    smallest = y 
else:
    smallest = z 

if x > y and x > z:
    largest = x 
elif y > z: 
    largest = y 
else: 
    largest = z 

print("Largest number is", largest)
print("Smallest number is", smallest)

if x == y or y == z or z == x:
    print("numbers are equal")
